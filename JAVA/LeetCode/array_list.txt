class Solution {
    public int searchInsert(int[] nums, int target) {
        int len = nums.length-1;
        if(target<nums[0])
            return 0;
        if(target>nums[len])
            return len+1;
        for(int i=0; i<nums.length; i++){
            if(nums[i]>=target){
                return i;
            }
        }
        return 0;
    }
}
